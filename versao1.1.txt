    <script>
        let dbMateriais = [];
        // Recupera progresso do navegador
        let concluidos = JSON.parse(localStorage.getItem('estudos_concluidos')) || [];

        // Tipos que aparecem no Check e na Barra de Progresso
        const tiposContaveis = ['prova', 'exercicio'];

        function configurarEtiqueta(tipo) {
            const tipos = {
                "prova": { label: "PROVA", color: "bg-red-50 text-red-600 border-red-100", icon: "fa-file-circle-exclamation" },
                "exercicio": { label: "EXERCÍCIO", color: "bg-blue-50 text-blue-600 border-blue-100", icon: "fa-list-check" },
                "material": { label: "MATERIAL", color: "bg-slate-100 text-slate-500 border-slate-200", icon: "fa-file" }
                /*"slide": { label: "SLIDE", color: "bg-amber-50 text-amber-600 border-amber-100", icon: "fa-display" },
                "atividade": { label: "ATIVIDADE", color: "bg-emerald-50 text-emerald-600 border-emerald-100", icon: "fa-pen-to-square" },*/
                
            };
            return tipos[tipo?.toLowerCase()] || tipos["material"];
        }

        function alternarConclusao(idArquivo) {
            const index = concluidos.indexOf(idArquivo);
            if (index === -1) {
                concluidos.push(idArquivo);
            } else {
                concluidos.splice(index, 1);
            }
            localStorage.setItem('estudos_concluidos', JSON.stringify(concluidos));
            renderizar(dbMateriais);
        }

        async function carregarDados() {
            try {
                const response = await fetch('materiais.json');
                dbMateriais = await response.json();
                renderizar(dbMateriais);
            } catch (e) {
                console.error(e);
                document.getElementById('container-materiais').innerHTML = `
                    <div class="text-center py-20 bg-white rounded-3xl border border-red-100 shadow-sm">
                        <i class="fa-solid fa-triangle-exclamation text-4xl text-red-500 mb-4"></i>
                        <p class="text-slate-700 font-bold">Erro ao carregar materiais.json</p>
                        <p class="text-slate-500 text-sm">Certifique-se de usar o Live Server no VS Code.</p>
                    </div>`;
            }
        }

        function renderizar(lista) {
            const container = document.getElementById('container-materiais');
            const menu = document.getElementById('menu-lateral');
            container.innerHTML = "";
            menu.innerHTML = "";

            // Agrupa por Matéria (Pasta)
            const agrupado = lista.reduce((acc, item) => {
                if (!acc[item.pasta]) acc[item.pasta] = [];
                acc[item.pasta].push(item);
                return acc;
            }, {});

            Object.keys(agrupado).sort().forEach(materia => {
                const arquivosMateria = agrupado[materia];
                
                // Filtra apenas Provas e Exercícios para o progresso
                const arquivosContaveis = arquivosMateria.filter(item => 
                    item.tipo && tiposContaveis.includes(item.tipo.toLowerCase())
                );
                
                const contagemConcluidos = arquivosContaveis.filter(item => 
                    concluidos.includes(`${materia}-${item.nome}`)
                ).length;
                
                const temProgresso = arquivosContaveis.length > 0;
                const porcentagem = temProgresso ? Math.round((contagemConcluidos / arquivosContaveis.length) * 100) : 0;

                // Adiciona ao Menu Lateral
                menu.innerHTML += `
                    <a href="#${materia}" class="px-4 py-2 rounded-xl text-sm font-medium hover:bg-indigo-800 transition-colors uppercase opacity-80 hover:opacity-100 whitespace-nowrap md:block">
                        • ${materia}
                    </a>`;

                // Cria Seção da Matéria
                let sectionHTML = `
                    <section id="${materia}" class="scroll-mt-24">
                        <div class="mb-8">
                            <div class="flex items-center justify-between mb-2">
                                <h2 class="text-2xl font-black text-slate-800 uppercase tracking-tighter">${materia}</h2>
                                ${temProgresso ? `<span class="text-xs font-bold text-indigo-600">${porcentagem}% Concluído</span>` : ''}
                            </div>
                            ${temProgresso ? `
                            <div class="w-full bg-slate-200 h-2 rounded-full overflow-hidden">
                                <div class="progress-bar bg-indigo-500 h-full" style="width: ${porcentagem}%"></div>
                            </div>` : '<div class="h-[1px] bg-slate-200 w-full"></div>'}
                        </div>
                        <div class="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-6">`;

                arquivosMateria.forEach(item => {
                    const idUnico = `${materia}-${item.nome}`;
                    const estaConcluido = concluidos.includes(idUnico);
                    const config = configurarEtiqueta(item.tipo);
                    const linkReal = encodeURI(`${item.pasta}/${item.nome}.${item.ext}`);
                    const podeMarcar = item.tipo && tiposContaveis.includes(item.tipo.toLowerCase());

                    sectionHTML += `
                        <div class="card-estudo bg-white p-6 rounded-3xl shadow-sm border border-slate-100 flex flex-col justify-between group ${estaConcluido && podeMarcar ? 'card-concluido' : ''}">
                            
                            ${podeMarcar ? `
                            <button onclick="alternarConclusao('${idUnico}')" class="absolute top-4 right-4 z-10 w-8 h-8 rounded-full border-2 transition-all flex items-center justify-center ${estaConcluido ? 'bg-green-500 border-green-500 text-white shadow-lg shadow-green-100' : 'bg-white border-slate-200 text-transparent hover:border-green-300'}">
                                <i class="fa-solid fa-check text-[10px]"></i>
                            </button>` : ''}
                            
                            <div>
                                <div class="flex items-start mb-4">
                                    <span class="text-[10px] font-bold px-3 py-1 rounded-md border ${config.color} uppercase tracking-widest">${config.label}</span>
                                </div>
                                <h3 class="font-bold text-slate-700 text-sm leading-tight mb-6 h-10 overflow-hidden ${estaConcluido && podeMarcar ? 'line-through opacity-50' : ''}">${item.nome}</h3>
                            </div>
                            
                            <a href="${linkReal}" target="_blank" 
                               class="flex items-center justify-center gap-2 w-full bg-indigo-600 text-white text-xs font-bold py-3.5 rounded-2xl hover:bg-indigo-700 transition-all shadow-md shadow-indigo-100 hover:shadow-indigo-200">
                               VISUALIZAR
                            </a>
                        </div>`;
                });
                sectionHTML += `</div></section>`;
                container.innerHTML += sectionHTML;
            });
        }

        // Filtro de Busca em tempo real
        document.getElementById('campoBusca').addEventListener('input', (e) => {
            const termo = e.target.value.toLowerCase();
            const filtrados = dbMateriais.filter(m => 
                m.nome.toLowerCase().includes(termo) || 
                m.pasta.toLowerCase().includes(termo)
            );
            renderizar(filtrados);
        });

        carregarDados();
    </script>
