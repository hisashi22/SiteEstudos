    <script>
        let dbMateriais = [];
        let concluidos = JSON.parse(localStorage.getItem('estudos_concluidos')) || [];
        const tiposContaveis = ['prova', 'exercicio'];

        async function carregarDados() {
            try {
                const response = await fetch('materiais.json');
                dbMateriais = await response.json();
                renderizarHome();
            } catch (e) { console.error("Erro ao carregar banco de dados."); }
        }

        function renderizarHome() {
            const grid = document.getElementById('grid-materias');
            grid.style.display = 'grid';
            document.getElementById('detalhe-materia').classList.remove('ativa');
            grid.innerHTML = "";

            const materias = [...new Set(dbMateriais.map(m => m.pasta.split(' / ')[0]))].sort();

            materias.forEach(m => {
                const isEnem = m.toLowerCase().includes('enem');
                const arquivosM = dbMateriais.filter(i => i.pasta.startsWith(m));
                const contaveis = arquivosM.filter(i => tiposContaveis.includes(i.tipo?.toLowerCase()));
                const feitos = contaveis.filter(i => concluidos.includes(gerarIdUnico(i))).length;
                const progresso = contaveis.length ? Math.round((feitos / contaveis.length) * 100) : 0;

                grid.innerHTML += `
                    <div onclick="verMateria('${m}')" class="card-materia p-8 rounded-3xl shadow-sm border border-slate-200 flex flex-col items-center text-center ${isEnem ? 'card-enem' : 'bg-white'}">
                        <div class="icon-box w-16 h-16 bg-indigo-50 rounded-2xl flex items-center justify-center text-indigo-600 mb-4 text-2xl transition-colors">
                            <i class="fa-solid ${isEnem ? 'fa-medal' : 'fa-book'}"></i>
                        </div>
                        <h3 class="font-bold text-slate-800 uppercase tracking-tight mb-2">${m}</h3>
                        <div class="w-full prog-bg bg-slate-100 h-2 rounded-full mt-4 overflow-hidden">
                            <div class="bg-indigo-500 h-full transition-all duration-700" style="width: ${progresso}%"></div>
                        </div>
                        <span class="text-[10px] font-bold opacity-60 mt-2 uppercase tracking-widest">${progresso}% concluído</span>
                    </div>`;
            });
        }

        function verMateria(materiaNome) {
            document.getElementById('grid-materias').style.display = 'none';
            const detalhe = document.getElementById('detalhe-materia');
            detalhe.classList.add('ativa');
            document.getElementById('titulo-materia-ativa').innerText = materiaNome;

            const listaAccordions = document.getElementById('lista-accordions');
            listaAccordions.innerHTML = "";

            const arquivosMateria = dbMateriais.filter(item => item.pasta.startsWith(materiaNome));
            const subcategorias = [...new Set(arquivosMateria.map(item => item.pasta.split(' / ')[1] || "Geral"))].sort();

            subcategorias.forEach(sub => {
                const arquivosSub = arquivosMateria.filter(item => (item.pasta.split(' / ')[1] || "Geral") === sub);
                const idSub = btoa(materiaNome + sub).replace(/=/g, '');

                listaAccordions.innerHTML += `
                    <div class="bg-white border border-slate-200 rounded-2xl overflow-hidden shadow-sm">
                        <button onclick="toggleAccordion('${idSub}')" class="w-full p-5 flex items-center justify-between hover:bg-slate-50 transition-colors">
                            <span class="font-bold text-slate-700">${sub}</span>
                            <div class="flex items-center gap-4">
                                <span class="text-[10px] font-bold bg-slate-100 text-slate-500 px-3 py-1 rounded-full uppercase">${arquivosSub.length} itens</span>
                                <i id="seta-${idSub}" class="fa-solid fa-chevron-down text-slate-300 seta"></i>
                            </div>
                        </button>
                        <div id="content-${idSub}" class="accordion-content border-t border-slate-50">
                            ${arquivosSub.map(arq => {
                                const idUnico = gerarIdUnico(arq);
                                const isDone = concluidos.includes(idUnico);
                                const podeCheck = tiposContaveis.includes(arq.tipo?.toLowerCase());
                                const idLimpo = btoa(idUnico).replace(/=/g, '');
                                return `
                                    <div class="flex items-center justify-between p-4 px-8 border-b border-slate-50 last:border-0 hover:bg-slate-50 transition-colors">
                                        <div class="flex items-center gap-4">
                                            ${podeCheck ? `
                                                <button onclick="marcar('${idUnico}', event)" id="btn-${idLimpo}" class="w-5 h-5 rounded border flex items-center justify-center transition-all ${isDone ? 'bg-indigo-600 border-indigo-600 text-white' : 'border-slate-300 bg-white text-transparent'}">
                                                    <i class="fa-solid fa-check text-[8px]"></i>
                                                </button>
                                            ` : '<i class="fa-solid fa-file text-slate-200"></i>'}
                                            <span id="txt-${idLimpo}" class="text-sm font-medium ${isDone && podeCheck ? 'line-through opacity-40 italic' : 'text-slate-600'}">${arq.nome}</span>
                                        </div>
                                        <a href="${encodeURI(arq.pasta + '/' + arq.nome + '.' + arq.ext)}" target="_blank" class="text-[10px] font-bold text-indigo-500 hover:text-indigo-700 uppercase">Abrir</a>
                                    </div>`;
                            }).join('')}
                        </div>
                    </div>`;
            });
        }

        function toggleAccordion(id) {
            const content = document.getElementById('content-' + id);
            const seta = document.getElementById('seta-' + id);
            content.classList.toggle('aberto');
            seta.classList.toggle('seta-rodada');
        }

        function marcar(idUnico, e) {
            e.stopPropagation();
            const idLimpo = btoa(idUnico).replace(/=/g, '');
            const btn = document.getElementById('btn-' + idLimpo);
            const txt = document.getElementById('txt-' + idLimpo);
            
            const index = concluidos.indexOf(idUnico);
            if (index === -1) {
                concluidos.push(idUnico);
                btn.className = "w-5 h-5 rounded border flex items-center justify-center transition-all bg-indigo-600 border-indigo-600 text-white";
                txt.classList.add('line-through', 'opacity-40', 'italic');
            } else {
                concluidos.splice(index, 1);
                btn.className = "w-5 h-5 rounded border border-slate-300 bg-white text-transparent flex items-center justify-center transition-all";
                txt.classList.remove('line-through', 'opacity-40', 'italic');
            }
            localStorage.setItem('estudos_concluidos', JSON.stringify(concluidos));
        }

        function gerarIdUnico(item) { return `${item.pasta}-${item.nome}`; }
        function voltarHome() { renderizarHome(); }

        // Busca rápida
        document.getElementById('campoBusca').addEventListener('input', (e) => {
            const termo = e.target.value.toLowerCase();
            if(termo === "") return renderizarHome();
            
            const filtrados = dbMateriais.filter(m => m.nome.toLowerCase().includes(termo));
            // Na busca, mostramos o detalhe diretamente
            document.getElementById('grid-materias').style.display = 'none';
            document.getElementById('detalhe-materia').classList.add('ativa');
            document.getElementById('titulo-materia-ativa').innerText = "Resultado da Busca";
            
            // Renderiza apenas uma lista simples de resultados
            const lista = document.getElementById('lista-accordions');
            lista.innerHTML = filtrados.map(arq => `
                <div class="bg-white p-4 px-8 border border-slate-200 rounded-xl mb-2 flex items-center justify-between">
                    <span class="text-sm font-medium text-slate-600">${arq.nome} <small class="text-indigo-400 ml-2">(${arq.pasta})</small></span>
                    <a href="${encodeURI(arq.pasta + '/' + arq.nome + '.' + arq.ext)}" target="_blank" class="text-[10px] font-bold text-indigo-500 uppercase">Abrir</a>
                </div>
            `).join('');
        });

        carregarDados();
    </script>
